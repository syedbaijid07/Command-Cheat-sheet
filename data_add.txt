# --- dalfox: automated xss scanner ---

tool: dalfox
command: dalfox url <url>
desc: scans a single url for xss vulnerabilities automatically
keywords: automated xss scan, dalfox scan url, find xss vulnerability, xss scanner

tool: dalfox
command: dalfox file <url-list-file>
desc: scans a list of urls from a file for xss vulnerabilities
keywords: bulk xss scan, dalfox multiple urls, scan url list for xss

tool: dalfox
command: dalfox url <url> --blind <interactsh-url>
desc: tests for blind xss by injecting a payload that calls back to an out-of-band server
keywords: blind xss testing, dalfox blind xss, out of band xss detection

tool: dalfox
command: dalfox url <url> -o <output-file>
desc: scans a url for xss and saves the findings to a file
keywords: save dalfox results, dalfox output file, export xss findings

tool: dalfox
command: dalfox pipe
desc: scans urls piped in from another tool (e.g. gau or waybackurls) for xss
keywords: dalfox pipe mode, chain dalfox with other tools, scan piped urls for xss

command: python3 jwt_tool.py <jwt-token> -T

keywords: crack jwt secret, jwt_tool crack mode, brute force jwt signing key
